in extanded_entity class constructor we will add two attributes.


				var extanded_entity = function(x,y,z,yimg)
				{
					this.type = 'extanded_entity'; 

					//extand entity
					y_entity.call(this,x,y,z,1,yimg);
					
					this.b_dir = "down";
					this.is_bullet_movment = true;
					

				}//end extanded_entity

first att holds the direction the entity will move.
the second one enables the use of bullet_movment.

no we will create the method move_to_do that will use the bullet movement method. 


				extanded_entity_p.update = function()
				{
					//do stuff when class created
					this.init();
					
					this.move_to_do();
					
					

					//call parent update func
					y_entity_p.update.call(this); 

				}//end update

				extanded_entity_p.move_to_do = function()
				{
					y_entity_p.bullet_movment.call(this);
				}//end alpha_do
